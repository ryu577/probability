```Python
import random
import numpy as np

array = []


for val in range(100000):
	number1 = random.uniform(0, 1)
	number2 = random.uniform(0, 1)


	if number1 < number2:
		length1 = number1
		length2 = number2-number1
		length3 = 1 - number2 
	else:
		length1 = number2
		length2 = number1 - number2 
		length3 = 1 - number1 


	if length1+length2 < length3 or length2+length3 < length1 or length1+length3 < length2:
		array.append(0)
	else:
		array.append(1)
print(np.mean(array))
```

#gamblersruin #probability 

```python
import numpy as np

n = 6  
p = 0.5  

PT = np.zeros((n-1, n-1))

for i in range(n):
    if i > 0:  
        PT[i, i-1] = 1 - p
    if i < n - 1:  
        PT[i, i+1] = p

I = np.eye(n)
I_minus_PT = I - PT
S = np.linalg.inv(I_minus_PT)

print(S[2][1])
```
